function xhrGetUno() {
    const id  = document.getElementById('xhr-id').value;
    const xhr = new XMLHttpRequest();

    xhr.open('GET', `api.php?id=${id}`, true);

    xhr.onload = function () {
        if (xhr.status >= 200 && xhr.status < 300) {
            const data = JSON.parse(xhr.responseText);
            mostrarResultado(data);
        } else {
            mostrarError(xhr.status, xhr.responseText);
        }
    };

    xhr.onerror = () => console.error('Error de red');
    xhr.send();
}
Fetch API con .then()

Promesas encadenadas. Notá cómo se verifica response.ok antes de leer el JSON.

Respuesta del servidor: — 
// Presioná un botón para ver la respuesta aquí
Ver código fuente 
function fetchGetTodos() {
    fetch('api.php')
        .then(response => {
            if (!response.ok) {
                throw new Error(`HTTP ${response.status}`);
            }
            return response.json();
        })
        .then(data => {
            console.log(data);
        })
        .catch(err => {
            console.error('Error:', err.message);
        });
}
async / await

La misma lógica que Fetch, pero con sintaxis que parece código síncrono. Más legible en operaciones en secuencia.

Respuesta del servidor: —

async function asyncGetTodos() {
    try {
        const res = await fetch('api.php');

        if (!res.ok) {
            throw new Error(`HTTP ${res.status}: ${res.statusText}`);
        }

        const data = await res.json();
        console.log(data);

    } catch (err) {
        console.error('Error:', err.message);
    }
}

// Buscar con parámetro de query
async function asyncBuscar() {
    const q = document.getElementById('async-q').value;
    const url = `api.php?q=${encodeURIComponent(q)}`;

    const res = await fetch(url);
    const data = await res.json();
    console.log(data);
}
